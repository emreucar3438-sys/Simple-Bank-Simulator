class BankAccount:
    """
    A class to manage bank account balances and store data in a local text file.
    """
    def __init__(self, balance, owner):
        self.balance = balance
        self.owner = owner
        # Load saved data when the application starts
        self.load_data()

    def deposit(self, amount):
        """Adds funds to the account and updates the storage."""
        self.balance += amount
        self.save_data()
        print(f"{amount} TL deposited successfully.")

    def withdraw(self, amount):
        """Subtracts funds if balance is sufficient and updates the storage."""
        if self.balance >= amount:
            self.balance -= amount
            self.save_data()
            print(f"{amount} TL withdrawn successfully.")
        else:
            print("Error: Insufficient balance.")

    def display_info(self):
        """Prints account details in a formatted way."""
        print(f"\n--- ACCOUNT INFORMATION ---")
        print(f"ACCOUNT OWNER: {self.owner}")
        print(f"CURRENT BALANCE: {self.balance} TL")

    def save_data(self):
        """Writes the current balance to a local txt file as a string."""
        with open("account_summary.txt", "w", encoding="utf-8") as file:
            file.write(str(self.balance))

    def load_data(self):
        """Reads balance from the file if it exists; otherwise, starts fresh."""
        try:
            with open("account_summary.txt", "r", encoding="utf-8") as file:
                content = file.read()
                if content:
                    self.balance = float(content)
        except (FileNotFoundError, ValueError):
            # If file is not found or empty, continue with the default balance
            pass

# --- MAIN PROGRAM LOOP ---
# Create an account object
account = BankAccount(0, "Emre")

while True:
    print("\n--- BANK MENU ---")
    print("1. Check Balance")
    print("2. Deposit Money")
    print("3. Withdraw Money")
    print("4. Exit")
    
    choice = input("Please select an action (1-4): ")

    if choice == "1":
        account.display_info()
    
    elif choice == "2":
        amount = float(input("Enter amount to deposit: "))
        account.deposit(amount)
        print("New balance:", account.balance)
    
    elif choice == "3":
        amount = float(input("Enter amount to withdraw: "))
        account.withdraw(amount)
    
    elif choice == "4":
        print("Exiting bank system. Have a nice day!")
        break 
    
    else:
        print("Invalid choice! Please enter a number between 1 and 4.")
